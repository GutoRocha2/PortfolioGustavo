class Conta:
    def __init__(self, _titular, numero):
        self.saldo = 0
        self.numero=numero
        self._titular = _titular

    @property
    def saldo(self):
        return self._saldo

    @saldo.setter
    def saldo(self, saldo):
        if (saldo<0):
            print("O saldo n√£o pode ser negativo")
        else:
            self._saldo = saldo
    def saque(self, valor):
        if(self.saldo>=valor):
            self.saldo-=valor
            print("Saque realizado com sucesso")
        else:
            print("Saldo insuficiente")

    def deposita(self, valor):
        self.saldo+=valor
